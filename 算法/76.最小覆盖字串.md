### 题解

class Solution {
    public String minWindow(String s, String t) {
        int minCount=Integer.MAX_VALUE;
        String minStr="";
        Map<Character,Integer> smap=new HashMap<>();
        Map<Character,Integer> tmap=new HashMap<>();
        for(int i=0; i<t.length(); i++){
            int count=tmap.getOrDefault(t.charAt(i),0);
            tmap.put(t.charAt(i),++count);
        }
        int l=0;
        int r=0;
        while(r<s.length()){
            int count=smap.getOrDefault(s.charAt(r),0);
            smap.put(s.charAt(r),++count);
            while(isMatch(smap,tmap) && l<=r){
                //保存快照
                if(r-l+1<minCount){
                    minCount=r-l+1;
                    minStr=s.substring(l,r+1);
                }
                //移除l
                count=smap.get(s.charAt(l));
                smap.put(s.charAt(l),--count);
                l++;
            }
            r++;
        }
        return minStr;
    }


    public boolean isMatch(Map<Character,Integer> smap, Map<Character,Integer> tmap){
        if(smap.size()<tmap.size()){
            return false;
        }
        for(Map.Entry<Character,Integer> entry:tmap.entrySet()){
            if(!smap.containsKey(entry.getKey())){
                return false;
            }
            if(smap.get(entry.getKey())<entry.getValue()){
                return false;
            }
        }
        return true;
    }
}
