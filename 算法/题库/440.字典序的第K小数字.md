字典序其实就是对这颗十叉树进行先序遍历：1，10，100...11，110，1100
```
class Solution {
    public int findKthNumber(int n, int k) {
        long p=1;
        long prefix=1;
        while(p<k){
            int count=getCount(prefix,n);
            //k在当前分支下，p继续深入，prefix也继续深入
            if(k < count+p){
                p++;
                prefix*=10;
			//k不在当前分支下，p和prefix都转到相邻的下一分支
            }else {
                p+=count;
                prefix++;
            }
        }
        return (int)prefix;

    }

	//1和2之间的个数为：1、2之间的个数+10、20之间的个数+100、200之间的个数...
    public int getCount(long prefix, int n){
        long count=0;
        long curr=prefix;
        long next=prefix+1;
        while(curr <= n){
			//当前层俩前后节点之间的个数
            count += (Math.min(n+1,next)-curr);
			//进入下一层
            curr*=10;
            next*=10;
        }
        return (int)count;
    }
}
```
