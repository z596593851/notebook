class Solution {
    public boolean isCovered(int[][] ranges, int left, int right) {
        int[]log=new int[52];
        for(int[]range:ranges){
            log[range[0]]++;
            log[range[1]+1]--;
        }
        int sum=0;
        for(int i=0; i<log.length; i++){
            sum+=log[i];
            if(i>=left && i<=right && sum<=0){
                return false;
            }
        }
        return true;
    }
}
