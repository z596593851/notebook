class Solution{

	public int packet(int n, int limit, int[]w, int[]v){
		int result=0;
		int dp[][]=new int[n+1][max+1];
		for(int i=1; i<=n; i++){
			for(int j=0; j<=limit; j++){
				if(j<wp[i-1]){
					dp[i][j]=dp[i-1][j];
				}else{
					dp[i][j]=Math.max(dp[i-1][j-w[i-1]]+v[i-1], dp[i-1][j]);
				}
				result=Math.max(result,dp[i][j]);
			}
		}
		return result;
	}
}
