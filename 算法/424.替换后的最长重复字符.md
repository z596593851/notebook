class Solution {
    public int characterReplacement(String s, int k) {
        int len=s.length();
        int maxCount=0;
        int result=0;
        char[]array=new char[26];
        int l=0, r=0;
        while(r<len){
            maxCount=Math.max(maxCount,++array[s.charAt(r++)-'A']);
            //不满足
            if((r-l)-maxCount>k){
                array[s.charAt(l)-'A']--;
                l++;
            }
        }
        return r-l;
    }
}
